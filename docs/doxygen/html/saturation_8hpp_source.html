249template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 64)>>
250constexpr uinteger_t<N> addu (uinteger_t<N> const x, uinteger_t<N> const y) {
251 assert (x <= ulimits<N>::max () && "addu<> x value out of range");
252 assert (y <= ulimits<N>::max () && "addu<> y value out of range");
253 constexpr auto maxu = mask_v<N>;
254 uinteger_t<N> res = x + y;
255 res |= -((res < x) | (res > maxu));
256 return res & maxu;
257}

273inline uinteger_t<N> addu_asm (uinteger_t<N> x, uinteger_t<N> y) {
274 uinteger_t<N> t;
275 __asm__(
276 "add %[y], %[x]\n\t" // x += y (sets carry C on overflow)
277 "sbb %[t], %[t]\n\t" // t -= t - C (t will become 0 or ~0).
278 "or %[t], %[x]\n\t" // x |= t
279 : [x] "+r"(x), [t] "=r"(t) // output

481template <typename T,
482 typename = typename std::enable_if_t<std::is_integral_v<T>>>
483constexpr std::pair<T, T> multiply (T const u, T const v) {
484 using UT = std::make_unsigned_t<T>;
485 constexpr auto half_shift =
486 sizeof (T) * CHAR_BIT / 2U; // half the number of bits in type T.
487 constexpr auto half_mask = (T{1} << half_shift) - 1U;
488
489 auto const u_hi = u >> half_shift;
490 auto const u_lo = static_cast<UT> (u) & half_mask;
491 auto const v_hi = v >> half_shift;
492 auto const v_lo = static_cast<UT> (v) & half_mask;
493
494 auto const lo_lo = u_lo * v_lo;
495 auto const lo_lo_mod = lo_lo & half_mask;
496 auto const lo_lo_over = lo_lo >> half_shift;
497
498 auto const hi_lo = static_cast<UT> (u_hi) * v_lo + lo_lo_over;
499 auto const hi_lo_mod = hi_lo & half_mask;
500 auto const hi_lo_over =
501 static_cast<UT> (static_cast<T> (hi_lo) >> half_shift);
502
503 auto const lo_hi = u_lo * static_cast<UT> (v_hi) + hi_lo_mod;
504 auto const lo_hi_over =
505 static_cast<UT> (static_cast<T> (lo_hi) >> half_shift);
506
507 auto const hi =
508 static_cast<T> (static_cast<UT> (u_hi * v_hi) + hi_lo_over + lo_hi_over);
509 auto const lo = static_cast<T> ((lo_hi << half_shift) +
510 lo_lo_mod); // (Expression could be "u * v")
511 return std::make_pair (hi, lo);
512}

516template <size_t N, bool IsUnsigned>
517struct multiplier {
519 using arg_type = std::conditional_t<IsUnsigned, uinteger_t<N>, sinteger_t<N>>;
520
527 constexpr std::pair<arg_type, arg_type> operator() (arg_type const x,
528 arg_type const y) const {
529 if constexpr (N <= narrow_multiply_max) {
530 auto const narrow_res =
531 static_cast<std::conditional_t<IsUnsigned, uinteger_t<N * 2>,
532 sinteger_t<N * 2>>> (x) *
533 y;
534 return std::make_pair (static_cast<arg_type> (narrow_res >> N),
535 static_cast<arg_type> (narrow_res));
536 } else {
537 return adjust (multiply (x, y));
538 }
539 }
540
541private:
542 static constexpr auto narrow_multiply_max = size_t{32};
543 static constexpr auto shift = sizeof (arg_type) * CHAR_BIT - N;
544 using unsigned_type = std::make_unsigned_t<arg_type>;
545
546 static constexpr std::pair<arg_type, arg_type> adjust (
547 std::pair<arg_type, arg_type>&& wide_res) {
548 if constexpr (shift == 0) {
549 return std::move (wide_res);
550 } else {
551 wide_res.first = static_cast<arg_type> (
552 static_cast<unsigned_type> (wide_res.first) << shift |
553 (static_cast<unsigned_type> (wide_res.second) >> N));
554 wide_res.second =
555 static_cast<arg_type> (static_cast<unsigned_type> (wide_res.second)
556 << shift) >>
557 shift;
558 assert (((wide_res.first >> N) == 0 ||
559 (!IsUnsigned &&
560 (wide_res.first >> N) == static_cast<arg_type> (-1))) &&
561 "multiplier<>-wide low result is out of range");
562 assert (((wide_res.second >> N) == 0 ||
563 (!IsUnsigned &&
564 (wide_res.second >> N) == static_cast<arg_type> (-1))) &&
565 "multiplier<>-wide high result is out of range");
566 return std::move (wide_res);
567 }
568 }
569};

663template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 64)>>
664constexpr sinteger_t<N> adds (sinteger_t<N> const x, sinteger_t<N> const y) {
665 assert (x >= slimits<N>::min () && x <= slimits<N>::max () &&
666 "adds<> x value out of range");
667 assert (y >= slimits<N>::min () && y <= slimits<N>::max () &&
668 "adds<> y value out of range");
669 using uint = uinteger_t<N>;
670 using sint = sinteger_t<N>;
671 using ubits = details::nbit_scalar<N, true>;
672 using sbits = details::nbit_scalar<N, false>;
673 // unsigned versions of x and y.
674 auto const ux = ubits{static_cast<uint> (x)};
675 auto const uy = ubits{static_cast<uint> (y)};
676 // Get the answer (with potential overflow).
677 auto const res = ubits{static_cast<uint> (ux + uy)};
678
679 // Calculate the overflowed result as max or min depending on the sign of ux.
680 auto const v = sbits{static_cast<sint> (
681 ubits{static_cast<uint> ((ux >> (N - 1U)) + slimits<N>::max ())})};
682 assert (v == (x < sint{0} ? slimits<N>::min () : slimits<N>::max ()));
683
684 // Check for overflow.
685 if (sbits{static_cast<sint> ((static_cast<uint> (v) ^ uy) | ~(uy ^ res))} >=
686 0) {
687 return v;
688 }
689 // There was no overflow: return the proper result.
690 return sbits{static_cast<sint> (res)};
691}
