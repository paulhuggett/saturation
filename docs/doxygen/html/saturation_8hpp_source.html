87template <size_t N, typename = typename std::enable_if_t<(N <= 64)>>
88struct mask {

107template <size_t N>
108struct limits {
110 using type = sinteger_t<N>;
113 static constexpr type max () {
114 return static_cast<type> ((uinteger_t<N>{1} << (N - 1U)) - 1);
115 }
118 static constexpr type min () { return static_cast<type> (-max () - 1); }
119};

189template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 64)>>
190constexpr uinteger_t<N> addu (uinteger_t<N> const x, uinteger_t<N> const y) {
191 assert (x <= ulimits<N>::max () && "addu<> x value out of range");
192 assert (y <= ulimits<N>::max () && "addu<> y value out of range");
193 constexpr auto maxu = mask_v<N>;
194 uinteger_t<N> res = x + y;
195 res |= -((res < x) | (res > maxu));
196 return res & maxu;
197}

378template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 32)>>
379constexpr uinteger_t<N> mulu (uinteger_t<N> const x, uinteger_t<N> const y) {
380 assert (x <= ulimits<N>::max () && "mulu<> x value out of range");
381 assert (y <= ulimits<N>::max () && "mulu<> y value out of range");
382 auto const res =
383 static_cast<uinteger_t<N * 2U>> (x) * static_cast<uinteger_t<N * 2U>> (y);
384 auto const hi = res >> N;
385 auto const lo = static_cast<uinteger_t<N>> (res);
386 return (lo | -!!hi) & mask_v<N>;
387}

452template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 64)>>
453constexpr sinteger_t<N> adds (sinteger_t<N> const x, sinteger_t<N> const y) {
454 assert (x >= limits<N>::min () && x <= limits<N>::max () &&
455 "adds<> x value out of range");
456 assert (y >= limits<N>::min () && y <= limits<N>::max () &&
457 "adds<> y value out of range");
458 using uint = uinteger_t<N>;
459 using sint = sinteger_t<N>;
460 using ubits = nbit_scalar<N, true>;
461 using sbits = nbit_scalar<N, false>;
462 // unsigned versions of x and y.
463 auto const ux = ubits{static_cast<uint> (x)};
464 auto const uy = ubits{static_cast<uint> (y)};
465 // Get the answer (with potential overflow).
466 auto const res = ubits{static_cast<uint> (ux + uy)};
467
468 // Calculate the overflowed result as max or min depending on the sign of ux.
469 auto const v = sbits{static_cast<sint> (
470 ubits{static_cast<uint> ((ux >> (N - 1U)) + limits<N>::max ())})};
471 assert (v == (x < sint{0} ? limits<N>::min () : limits<N>::max ()));
472
473 // Check for overflow.
474 if (sbits{static_cast<sint> ((static_cast<uint> (v) ^ uy) | ~(uy ^ res))} >=
475 0) {
476 return v;
477 }
478 // There was no overflow: return the proper result.
479 return sbits{static_cast<sint> (res)};
480}

542template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 64)>>
543constexpr sinteger_t<N> subs (sinteger_t<N> const x, sinteger_t<N> const y) {
544 assert (x >= limits<N>::min () && x <= limits<N>::max () &&
545 "subs<> x value out of range");
546 assert (y >= limits<N>::min () && y <= limits<N>::max () &&
547 "subs<> y value out of range");
548 using uint = uinteger_t<N>;
549 using sint = sinteger_t<N>;
550 using ubits = nbit_scalar<N, true>;
551 using sbits = nbit_scalar<N, false>;
552
553 // unsigned versions of x and y.
554 auto const ux = ubits{static_cast<uint> (x)};
555 auto const uy = ubits{static_cast<uint> (y)};
556 auto const res = ubits{static_cast<uint> (ux - uy)};
557
558 // Check for overflow: ux must be different from uy and res.
559 if (sbits{static_cast<sint> ((ux ^ uy) & (ux ^ res))} < 0) {
560 // Calculate the overflowed result as max or min depending on the sign of
561 // ux.
562 auto const v = sbits{static_cast<sint> (
563 ubits{static_cast<uint> ((ux >> (N - 1U)) + limits<N>::max ())})};
564 assert (v == (x < sint{0} ? limits<N>::min () : limits<N>::max ()));
565 return v;
566 }
567 // There was no overflow: return the proper result.
568 return sbits{static_cast<sint> (res)};
569}

710template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 32)>>
711constexpr sinteger_t<N> muls (sinteger_t<N> const x, sinteger_t<N> const y) {
712 assert (x >= limits<N>::min () && x <= limits<N>::max () &&
713 "muls<> x value out of range");
714 assert (y >= limits<N>::min () && y <= limits<N>::max () &&
715 "muls<> y value out of range");
716
717 using s2bits = nbit_scalar<N * 2, false>;
718 using s2int = typename s2bits::type;
719
720 using sbits = nbit_scalar<N, false>;
721 using sint = typename sbits::type;
722 using ubits = nbit_scalar<N, true>;
723 using uint = typename ubits::type;
724
725 auto const res = s2bits{
726 static_cast<s2int> (static_cast<s2int> (x) * static_cast<s2int> (y))};
727 if (static_cast<sint> (res >> N) != static_cast<sint> (res >> (N - 1U))) {
728 auto const v = sbits{static_cast<sint> (ubits{static_cast<uint> (
729 static_cast<uint> (ubits{static_cast<uint> (x ^ y)} >> (N - 1U)) +
730 static_cast<uint> (limits<N>::max ()))})};
731 assert (v == (res < sint{0} ? limits<N>::min () : limits<N>::max ()));
732 return v;
733 }
734 return static_cast<sint> (res);
735}
