249template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 64)>>
250constexpr uinteger_t<N> addu (uinteger_t<N> const x, uinteger_t<N> const y) {
251 assert (x <= ulimits<N>::max () && "addu<> x value out of range");
252 assert (y <= ulimits<N>::max () && "addu<> y value out of range");
253 constexpr auto maxu = mask_v<N>;
254 uinteger_t<N> res = x + y;
255 res |= -((res < x) | (res > maxu));
256 return res & maxu;
257}

273inline uinteger_t<N> addu_asm (uinteger_t<N> x, uinteger_t<N> y) {
274 uinteger_t<N> t;

276 "add %[y], %[x]\n\t" // x += y (sets carry C on overflow)

473template <typename T,
474 typename = typename std::enable_if_t<std::is_integral_v<T>>>
475constexpr std::pair<T, T> multiply (T const u, T const v) {
476 using UT = std::make_unsigned_t<T>;
477 constexpr auto half_shift =
478 sizeof (T) * CHAR_BIT / 2U; // half the number of bits in type T.
479 constexpr auto half_mask = (T{1} << half_shift) - 1U;
480
481 auto const u_hi = u >> half_shift;
482 auto const u_lo = static_cast<UT> (u) & half_mask;
483 auto const v_hi = v >> half_shift;
484 auto const v_lo = static_cast<UT> (v) & half_mask;
485
486 auto const lo_lo = u_lo * v_lo;
487 auto const lo_lo_mod = lo_lo & half_mask;
488 auto const lo_lo_over = lo_lo >> half_shift;
489
490 auto const hi_lo = static_cast<UT> (u_hi) * v_lo + lo_lo_over;
491 auto const hi_lo_mod = hi_lo & half_mask;
492 auto const hi_lo_over =
493 static_cast<UT> (static_cast<T> (hi_lo) >> half_shift);
494
495 auto const lo_hi = u_lo * static_cast<UT> (v_hi) + hi_lo_mod;
496 auto const lo_hi_over =
497 static_cast<UT> (static_cast<T> (lo_hi) >> half_shift);
498
499 auto const hi =
500 static_cast<T> (static_cast<UT> (u_hi * v_hi) + hi_lo_over + lo_hi_over);
501 auto const lo = static_cast<T> ((lo_hi << half_shift) +
502 lo_lo_mod); // (Expression could be "u * v")
503 return std::make_pair (hi, lo);
504}

506template <size_t N, bool IsUnsigned>
507struct multiplier {
508 using arg_type = std::conditional_t<IsUnsigned, uinteger_t<N>, sinteger_t<N>>;
509
510 constexpr std::pair<arg_type, arg_type> operator() (arg_type const x,
511 arg_type const y) const {
512 if constexpr (N <= narrow_multiply_max) {
513 auto const narrow_res =
514 static_cast<std::conditional_t<IsUnsigned, uinteger_t<N * 2>,
515 sinteger_t<N * 2>>> (x) *
516 y;
517 return std::make_pair (static_cast<arg_type> (narrow_res >> N),
518 static_cast<arg_type> (narrow_res));
519 } else {
520 return adjust (multiply (x, y));
521 }
522 }
523
524private:
525 static constexpr auto narrow_multiply_max = size_t{32};
526 static constexpr auto shift = sizeof (arg_type) * CHAR_BIT - N;
527 using unsigned_type = std::make_unsigned_t<arg_type>;
528
529 static constexpr std::pair<arg_type, arg_type> adjust (
530 std::pair<arg_type, arg_type>&& wide_res) {
531 if constexpr (shift == 0) {
532 return std::move (wide_res);
533 } else {
534 wide_res.first = static_cast<arg_type> (
535 static_cast<unsigned_type> (wide_res.first) << shift |
536 (static_cast<unsigned_type> (wide_res.second) >> N));
537 wide_res.second =
538 static_cast<arg_type> (static_cast<unsigned_type> (wide_res.second)
539 << shift) >>
540 shift;
541 assert (((wide_res.first >> N) == 0 ||
542 (!IsUnsigned &&
543 (wide_res.first >> N) == static_cast<arg_type> (-1))) &&
544 "multiplier<>-wide low result is out of range");
545 assert (((wide_res.second >> N) == 0 ||
546 (!IsUnsigned &&
547 (wide_res.second >> N) == static_cast<arg_type> (-1))) &&
548 "multiplier<>-wide high result is out of range");
549 return std::move (wide_res);
550 }
551 }
552};

646template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 64)>>
647constexpr sinteger_t<N> adds (sinteger_t<N> const x, sinteger_t<N> const y) {
648 assert (x >= limits<N>::min () && x <= limits<N>::max () &&
649 "adds<> x value out of range");
650 assert (y >= limits<N>::min () && y <= limits<N>::max () &&
651 "adds<> y value out of range");
652 using uint = uinteger_t<N>;
653 using sint = sinteger_t<N>;
654 using ubits = details::nbit_scalar<N, true>;
655 using sbits = details::nbit_scalar<N, false>;
656 // unsigned versions of x and y.
657 auto const ux = ubits{static_cast<uint> (x)};
658 auto const uy = ubits{static_cast<uint> (y)};
659 // Get the answer (with potential overflow).
660 auto const res = ubits{static_cast<uint> (ux + uy)};
661
662 // Calculate the overflowed result as max or min depending on the sign of ux.
663 auto const v = sbits{static_cast<sint> (
664 ubits{static_cast<uint> ((ux >> (N - 1U)) + limits<N>::max ())})};
665 assert (v == (x < sint{0} ? limits<N>::min () : limits<N>::max ()));
666
667 // Check for overflow.
668 if (sbits{static_cast<sint> ((static_cast<uint> (v) ^ uy) | ~(uy ^ res))} >=
669 0) {
670 return v;
671 }
672 // There was no overflow: return the proper result.
673 return sbits{static_cast<sint> (res)};
674}
