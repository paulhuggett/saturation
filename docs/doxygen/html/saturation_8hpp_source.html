107template <size_t N>
108struct limits {
110 using type = sinteger_t<N>;
113 static constexpr type max () {
114 return static_cast<type> ((uinteger_t<N>{1} << (N - 1U)) - 1);

190constexpr uinteger_t<N> addu (uinteger_t<N> const x, uinteger_t<N> const y) {
191 assert (x <= ulimits<N>::max () && "addu<> x value out of range");
192 assert (y <= ulimits<N>::max () && "addu<> y value out of range");
193 constexpr auto maxu = mask_v<N>;
194 uinteger_t<N> res = x + y;
195 res |= -((res < x) | (res > maxu));

375template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 32)>>
376constexpr uinteger_t<N> mulu (uinteger_t<N> const x, uinteger_t<N> const y) {
377 assert (x <= ulimits<N>::max () && "mulu<> x value out of range");
378 assert (y <= ulimits<N>::max () && "mulu<> y value out of range");

449template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 64)>>
450constexpr sinteger_t<N> adds (sinteger_t<N> const x, sinteger_t<N> const y) {
451 assert (x >= limits<N>::min () && x <= limits<N>::max () &&
452 "adds<> x value out of range");
453 assert (y >= limits<N>::min () && y <= limits<N>::max () &&
454 "adds<> y value out of range");
455 using uint = uinteger_t<N>;
456 using sint = sinteger_t<N>;
457 using ubits = nbit_scalar<N, true>;
458 using sbits = nbit_scalar<N, false>;
459 // unsigned versions of x and y.
460 auto const ux = ubits{static_cast<uint> (x)};
461 auto const uy = ubits{static_cast<uint> (y)};
462 // Get the answer (with potential overflow).
463 auto const res = ubits{static_cast<uint> (ux + uy)};
464
465 // Calculate the overflowed result as max or min depending on the sign of ux.
466 auto const v = sbits{static_cast<sint> (
467 ubits{static_cast<uint> ((ux >> (N - 1U)) + limits<N>::max ())})};
468 assert (v == (x < sint{0} ? limits<N>::min () : limits<N>::max ()));
469
470 // Check for overflow.
471 if (sbits{static_cast<sint> ((static_cast<uint> (v) ^ uy) | ~(uy ^ res))} >=
472 0) {
473 return v;
474 }
475 // There was no overflow: return the proper result.
476 return sbits{static_cast<sint> (res)};
477}

540constexpr sinteger_t<N> subs (sinteger_t<N> const x, sinteger_t<N> const y) {

542 "subs<> x value out of range");
543 assert (y >= limits<N>::min () && y <= limits<N>::max () &&
544 "subs<> y value out of range");
545 using uint = uinteger_t<N>;
546 using sint = sinteger_t<N>;
547 using ubits = nbit_scalar<N, true>;
548 using sbits = nbit_scalar<N, false>;
549
550 // unsigned versions of x and y.
551 auto const ux = ubits{static_cast<uint> (x)};
552 auto const uy = ubits{static_cast<uint> (y)};
553 auto const res = ubits{static_cast<uint> (ux - uy)};
554
555 // Calculate the overflowed result as max or min depending on the sign of ux.
556 auto const v = sbits{static_cast<sint> (

558 assert (v == (x < sint{0} ? limits<N>::min () : limits<N>::max ()));

560 // Check for overflow: ux must be different from uy and res.
561 if (sbits{static_cast<sint> ((ux ^ uy) & (ux ^ res))} < 0) {
562 return v;

565 return sbits{static_cast<sint> (res)};

706template <size_t N, typename = typename std::enable_if_t<(N >= 4 && N <= 32)>>
707constexpr sinteger_t<N> muls (sinteger_t<N> const x, sinteger_t<N> const y) {
708 assert (x >= limits<N>::min () && x <= limits<N>::max () &&
709 "muls<> x value out of range");
710 assert (y >= limits<N>::min () && y <= limits<N>::max () &&
711 "muls<> y value out of range");
712
713 using s2bits = nbit_scalar<N * 2, false>;
714 using s2int = typename s2bits::type;
715
716 using sbits = nbit_scalar<N, false>;
717 using sint = typename sbits::type;
718 using ubits = nbit_scalar<N, true>;
719 using uint = typename ubits::type;
720
721 auto const res = s2bits{
722 static_cast<s2int> (static_cast<s2int> (x) * static_cast<s2int> (y))};
723 if (static_cast<sint> (res >> N) != static_cast<sint> (res >> (N - 1U))) {
724 auto const v = sbits{static_cast<sint> (ubits{static_cast<uint> (
725 static_cast<uint> (ubits{static_cast<uint> (x ^ y)} >> (N - 1U)) +
726 static_cast<uint> (limits<N>::max ()))})};
727 assert (v == (res < sint{0} ? limits<N>::min () : limits<N>::max ()));
728 return v;
729 }
730 return static_cast<sint> (res);
731}
